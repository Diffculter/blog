# 函数式编程

参考原文来源: 掘金

# 在做不同的事情的情况下，可能是基于某一个相同的条件

// 我要做2+3、2+4的操作

// 传统写法
const a = 2;
const b = 3;
const c = 4;
a + b || a + c

// 函数式写法
const calc = a => b => a + b
const addA = calc(2)
addA(3) || addA(4)

{
    //解释
    //1.0
    const calc = (a) => {
        return (b) => a + b;
    }
    //2.0
    const addA = (b) => 2 + b;
    //3.0
    addA(3) || addA(4);
}

# 不要让你的函数，去影响到外部的变量

// 非函数式
const a = 1;
function calc(){
    ++a
}
calc()

// 函数式
const a = 1;
function calc(num){
    return ++num
}
calc(a)
